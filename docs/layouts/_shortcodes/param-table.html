{{/* param-table: Show a table of parameters for a page. Usage: {{< param-table ignoreParams="skipme" >}} */}} {{- $ignoreParams := .Get "ignoreParams" | default "" -}} {{- $ignoreParamsList := split $ignoreParams "," -}} {{- $ignoreParamsList = append $ignoreParamsList ( slice "title" "weight" "draft" "iscjklanguage" "date" ) -}} {{- $params := .Page.Params }} {{- if .Page.IsHome }} {{- $params = .Site.Params }} {{- end }} {{- range $key, $value := $params }} {{- if not ( in $ignoreParamsList $key ) }} {{- end }} {{- end }}
{{ $key }} {{ $value }}
